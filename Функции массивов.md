###  Поиск по массивам
```php
<?php
$arr = [1, 2, 3, 4, 5];
print_r(in_array(3, $arr));
?>

<?php
$arr = ['a', '-', 'b',
        '-', 'c',
        '-', 'd'];
echo array_search('-', $arr);
?>

<?php
$arr = ['a', '-', 'b',
        '-', 'c',
        '-', 'd'];
$delete = array_search("-", $arr);
array_splice($arr, $delete, 1);
print_r($arr);
?>
```

### Граничные элементы
```php
<?php
$arr = [1, 2, 3, 4, 5];
print_r(array_shift($arr));
print_r(array_pop($arr));
?>

<?php
$arr = [1, 2, 3, 4, 5];
array_unshift($arr, "0");
array_push($arr, "6");
print_r($arr);
?>
```

### Сплайсинг массивов
```php
<?php

?>

<?php
$arr = [1, 2, 3, 4, 5];

```


### Слияние массивов
```php
<?php
$arr1 = [1, 2, 3];
$arr2 = ['a', 'b', 'c'];
$res = array_merge($arr1,
        $arr2);
print_r($res);
?>

<?php
$arr1 = ['a', 'b', 'c'];
$arr2 = [1, 2, 3];
$res = array_combine($arr1,
        $arr2);
print_r($res);
?>
```

### Сортировка массивов
```php
<?php

?>

ksort($arr);
print_r($arr);
?>

<?php

krsort($arr);
print_r($arr);
?>

<?php
$arr = ['a'=>5, 'b'=>2, 'c'=>4, 'd'=>1, 'e'=>3];

```

### Множественные операции
```php
<?php
$arr1 = [1, 2, 3, 4, 5];
$arr2 = [3, 4, 5, 6, 7];
$res = array_intersect($arr1, $arr2);
print_r($res);
?>

<?php
$arr1 = [1, 2, 3, 4, 5];
$arr2 = [3, 4, 5, 6, 7];
$res1 = array_diff($arr1, $arr2);
$res2 = array_diff($arr2, $arr1);
$result = array_merge($res1, $res2);
print_r($result);
?>
```
