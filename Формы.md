### Основы работы с формами
```php
<form action="/result.php">
    <input name="AAA">
    <input age="34">
    <input salary="2000">
    <input type="submit">
</form>
```

### Метод отправки формы
```php
<form action="/result.php" method="GET">
    <input name="name">
    <input name="age">
    <input name="salary">
    <input type="submit">
</form>

<form action="/result.php" method="POST">
    <input name="name">
    <input name="age">
    <input name="salary">
    <input type="submit">
</form>
```

### Получение данных форм методом GET
```php
<form action="/" method="GET">
    <input name="1">
    <input name="2">
    <input name="3">
    <input type="submit">
</form>
<?php
echo $_GET['1'] + $_GET['2'] + $_GET['3'];
?>
```

### Получение данных форм методом POST
```php
<form action="/" method="POST">
    <input name="name">
    <input name="age">
    <input type="submit">
</form>
<?php
echo 'name: ' . $_POST['name'] . ' age: ' .  $_POST['age'];
?>

<form action="/" method="POST">
    <input type="password" name="pass">
    <input type="submit">
</form>
<?php
$pass = '12345';
if ($_POST['pass'] == $pass) {
    echo 'Password is correct';
} else {
    echo 'Password is wrond! Try again';
}
?>

<form action="/" method="POST">
    <input name="date">
    <input name="month">
    <input name="year">
    <input type="submit">
</form>
<?php
$birthday = mktime(0, 0, 0, $_POST['month'],$_POST['day'], $_POST['year']);
$birthday_weekday = date('w', $birthday);
$weekdays = array(
    1 => 'Monday',
    2 => 'Tuesday',
    3 => 'Wednesday',
    4 => 'Thursday',
    5 => 'Friday',
    6 => 'Saturday',
    0 => 'Sunday'
);
foreach ($weekdays as $key => $day) {
    if ((int)$birthday_weekday === $key) {
        echo $day;
    }
}
?>
```

### Обработка формы в одном файле
```php
<form action="" method="GET">
    <input name="name">
    <input name="surname">
    <input name="patronymic">
    <input type="submit">
</form>
<?php
    echo $_GET['name'] . ' ' . $_GET['surname'] . ' ' . $_GET['patronymic'];
?>
```

### Скрытие формы после отправки
<?php
if (empty($_GET)) {
    ?>
    <form action="" method="GET">
        <input name="name">
        <input type="submit">
    </form>
    <?php
} else {
    echo 'Здравствуйте, ' . $_GET['name'];
}
?>

### Сохранение значений формы после отправки
```php
<form action="" method="GET">
    <input name="city"
        value="<?php if (isset($_GET['city']))
            echo $_GET['city'] ?>"
    >
    <input name="country"
           value="<?php if (isset($_GET['country']))
               echo $_GET['country'] ?>"
    >
    <input type="submit">
</form>
<?php
echo $_GET['city'] . ' ' . $_GET['country'];
?>
```

### Сохранение значения по умолчанию формы
```php
<form action="" method="GET">
    <input name="year"
        value="<?php
        if (isset($_GET['year']))
            echo $_GET['year'];
        else echo date('Y')
        ?>"
    >
    <input type="submit">
</form>
<?php
if (($_GET['year'] % 4 == 0) && ($_GET['year'] % 100 != 0)) {
    echo 'Високосный';
} else {
    echo 'Не високосный';
}
?>
```

### Сокращенный код для сохранения значений по умолчанию
```php
<form action="" method="POST">
    <input name="year" value="<?=
        $_GET['year'] ?? date('Y') ?>">
    <input name="month" value="<?=
        $_GET['month'] ?? date('m') ?>">
    <input name="day" value="<?=
        $_GET['day'] ?? date('d') ?>">
    <input type="submit">
</form>
<?php
//var_dump($_POST['year']);die;
$date = mktime(0, 0, 0, $_POST['month'],$_POST['day'], $_POST['year']);
$nextNewYear = strtotime("1 january + 1 year");
echo round(($nextNewYear - $date) / 86400);
?>
```
