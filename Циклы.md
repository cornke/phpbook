```php
<?php
$arr = [1, 2, 3, 4, 5];
foreach ($arr as $elem) {
    echo $elem ** 2;
}
?>

<?php

    echo "$elem<br>";
}
?>

    }
```

```php
    if ($elem > 0) {
        $sum += $elem;
    }

        echo $elem. ' ';
    }

<?php

    if ($key <= 4) {
        echo $day.' ';

        echo "<b>$day</b>".' ';
    }

<?php

    }
}
?>
```

### Работа с флагами

```php
        $flag = true;

    echo '+++';
} else {
    echo '---';
}
```
