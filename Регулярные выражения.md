### Замена с коллбэком через регулярки
```php
<?php
$str = 'The numbers are 3, 7 and 12';
$res = preg_replace_callback('#(\d+).*?(\d+).*?(\d+)#', function($match) {
    return 'The numbers are '.($match[1]**2).', '.($match[2]**2).' and '.($match[3]**2);
}, $str);
echo $res;
?>

<?php
$str = '2023-01-15 2022-12-31 2024-05-20';
$res = preg_replace_callback('#(\d{4})-(\d{2})-(\d{2})#', function($match) {
    return ($match[3]).'.'.($match[2]).'.'.($match[1]);
}, $str);
echo $res;
?>

<?php
$str = 'Costs: $15, $20, $100';
$res = preg_replace_callback('#(\d+).*?(\d+).*?(\d+)#', function($match) {
    return "Costs: $".($match[1]*1.1).", $".($match[2]*1.1).", $".($match[3]*1.1);
}, $str);
echo $res;
?>
```
