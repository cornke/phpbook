### Проверка строк
```php
<?php
$str = 'document.pdf';
echo str_ends_with($str, ".pdf");
?>

<?php
$str = 'https://example.com';
echo str_starts_with($str, "https://");
?>

<?php
$str = 'The quick brown fox';
echo str_contains($str, "fox");
?>

<?php
$str = 'image.png';
if ((str_ends_with($str, ".png")) || (str_ends_with($str, ".jpg"))) {
    echo true;
}
?>

<?php

    echo true;
}
?>
```

### Поиск по строкам
```php
<?php
$str = 'abc abc abc';
echo strpos($str, 'b');
?>

<?php
$str = 'abc abc abc';
echo strrpos($str, 'b');
?>

<?php
$str = 'abc abc abc';
echo stripos($str, 'b', 3);
?>

<?php
$str = 'aaa aaa aaa aaa aaa';
echo stripos($str, ' ', 6);
?>

<?php
$str = 'example..com';
$dotPosition = strpos($str, '.');
if (isset($str[$dotPosition + 1])
        && $str[$dotPosition + 1] == '.') {
    echo true;
}
echo false;
?>

<?php
$str = 'http://example.com';
if (stripos($str, 'http://') == 0) {
    echo true;
}
?>
```
