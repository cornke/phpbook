### Флеш сообщения
```php
<?php
session_start();

$_SESSION['flash'] = 'message';
header('Location: index.php');
die();

<?php
session_start();

if (isset($_SESSION['flash'])) {
    echo $_SESSION['flash'];
    unset($_SESSION['flash']);
}
```

### Массив флеш сообщений
```php
<?php
session_start();
$_SESSION['flash'][] = 'cat';
$_SESSION['flash'][] = 'says';
$_SESSION['flash'][] = 'meow';
if (!empty($_SESSION['flash'])) {
    foreach ($_SESSION['flash'] as $flash) {
        echo $flash;
    }

    $_SESSION['flash'] = [];
} else {
    echo '';
}
```

### Отправка формы в БД и редирект
```php
<form method="POST">
    <input name="test1">
    <input name="test2">
    <input type="submit">
</form>
<meta charset="utf-8">
<?php
/**
 * @var $link mysqli
 */
require_once("connect.php");
session_start();
mysqli_query($link, 'select * from user');

if (!empty($_POST)) {
    $_SESSION['flash'] = 'форма успешно сохранена';
    mysqli_query($link, "update user set name={$_POST['test1']}, surname={$_POST['test2']} where id = 1");
    header('Location: index.php');
    die();
}
if (isset($_SESSION['flash'])) {
    echo $_SESSION['flash'];
    unset($_SESSION['flash']);
}
```

### Редирект при валидации формы
//1 и 2
```php
<form method="POST">
    <input name="test1">
    <input name="test2">
    <input type="submit">
</form>
<meta charset="utf-8">
<?php
/**
 * @var $link mysqli
 */
require_once("connect.php");
session_start();
mysqli_query($link, 'select * from user');

if (!empty($_POST)) {
    if (валидация формы) {
        // сохраняем в базу
        $_SESSION['flash'] = 'форма успешно сохранена';
        header('Location: form.php');
        die();
        }
    } else {
        $_SESSION['flash'] = 'форма не прошла валидацию';
    }
    mysqli_query($link, "update user set name={$_POST['test1']}, surname={$_POST['test2']} where id = 1");
    header('Location: index.php');
}
if (isset($_SESSION['flash'])) {
    echo $_SESSION['flash'];
    unset($_SESSION['flash']);
}
```
